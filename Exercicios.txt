    Exercício 31 Pedro Henrique Ferrari
    
    static void Main(string[] args) {
      int a, x, par =0, impar =0, positivo =0, negativo=0;

      do{
        Console.Clear();
        Console.WriteLine("Digite um numero");
        a = int.Parse(Console.ReadLine());

        if(a > 0){
          positivo++;
        }
        else{
          negativo++;
        }

        if(a % 2 == 0){
          par++;
        }
        else{
          impar++;
        }
        Console.WriteLine("Deseja fechar o program (sim = 1, não = 0)");
        x = int.Parse(Console.ReadLine());
      }while(x == 0);

      Console.WriteLine("Numero de pares: {0}",par);
      Console.WriteLine("Numero de impares: {0}", impar);
      Console.WriteLine("Numero de positivos: {0}", positivo++);
      Console.WriteLine("Numero de negativos: {0}", negativo++);

    }
}
